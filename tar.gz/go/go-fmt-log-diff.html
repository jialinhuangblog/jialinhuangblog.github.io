TL;DR

  1. log will print time too.
  1. log also has
    1. Fatal: print and exit(1)

  1. log doesn’t have

print, printf, println

  • print: no format needs
  • println: no format and add a new line at the end.
  • printf: work with format specifiers

format specifiers

%dint, int8, int16, int32, int64decimal
%Tanytype of variable
%vanydefault, suit for all
%e, %f, %g, efgfloat32, float64floating-point formats
%qjust wrap with quote sign
%sstringstring
%tboolboolean
%ppointerpointer address

%v, %+v, %#v

Without String() method

type Person struct {
	Name string
	Age  int
}

func main() {
    p := Person{Name: "Alice", Age: 30}
    fmt.Printf("%v\n", p)   // {Alice 30}
    fmt.Printf("%+v\n", p)  // {Name:Alice Age:30}
    fmt.Printf("%#v\n", p)  // main.Person{Name:"Alice", Age:30}
}

With String() method

type Person struct {
	Name string
	Age  int
}

func (p Person) String() string {
	return fmt.Sprintf("customized(Name: %s, Age: %d)", p.Name, p.Age)
}

func main() {
    p := Person{Name: "Alice", Age: 30}
    fmt.Printf("%v\n", p)   // customized(Name: Alice, Age: 30)
    fmt.Printf("%+v\n", p)  // customized(Name: Alice, Age: 30)
    fmt.Printf("%#v\n", p)  // main.Person{Name:"Alice", Age:30}
}

Note: %v and %+v are affected by the String() method implementation

%v will check if the value implements the String() method

https://pkg.go.dev/fmt#Stringer

type Stringer interface {
	String() string
}

F Series (Fprint, Fprintln, Fprintf)

to specific destinations

func main() {
	// to ternimal
	fmt.Fprintf(os.Stdout, "Hello, %s!\n", "world")
	
	// to a file
	file, _ := os.Create("output.txt")
	defer file.Close()
	fmt.Fprintln(file, "This is written to a file.")
}

S Series (Sprint, Sprintln, Sprintf)

Used for concatenating strings

str := fmt.Sprint(arr)
fmt.Println(str) // output: [1 2 3 4]
